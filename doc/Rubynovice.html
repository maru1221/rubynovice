def DistanceSum(route,city)
  i=0
  sum=0
  while route.length > i+1 do
    sum += my_distance(city[route[i]],city[route[i+1]])
    #    puts my_distance(city[route[i]],city[route[i+1]])                          
    i+=1
    #    puts 1                                                                     
  end
  return sum
end

def my_distance(x1,x2)
  sq0=(x1[0]-x2[0])**2
  sq1=(x1[1]-x2[1])**2
  sq=sq0+sq1

  return Math::sqrt(sq)
end
